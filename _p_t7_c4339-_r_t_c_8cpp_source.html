92uint8_t PT7C4339::begin()
93{
94 _i2cWire->beginTransmission( _i2cAddress );
95 uint8_t error = _i2cWire->endTransmission();
96
97 if( error != 0 )
98 {
99 if( _SDA == 0 && _SCL == 0 )
100 {
101 _i2cWire->begin();
102 _i2cWire->setClock( _frequency );
103 }
104 else
105 {
106 _i2cWire->begin( _SDA, _SCL );
107 _i2cWire->setClock( _frequency );
108 }
109
110 _i2cWire->beginTransmission( _i2cAddress );
111 error = _i2cWire->endTransmission();
112
113 if( error != 0 ) return 0;
114 }
115
116 uint8_t hours = readRegister( PT7C4339_REG_HOURS );
117 bool is12H = hours & 0x40;
118 if( is12H )
119 {
120 hours &= 0xBF;
121 if( !writeRegister( PT7C4339_REG_HOURS, hours ) ) return 0;
122 }
123
124 uint8_t hoursA1 = readRegister( PT7C4339_REG_A1_HOURS );
125 bool A1Is12H = hoursA1 & 0x40;
126 if( A1Is12H )
127 {
128 hoursA1 &= 0xBF;
129 if( !writeRegister( PT7C4339_REG_A1_HOURS, hoursA1 ) ) return 0;
130 }
131
132 uint8_t hoursA2 = readRegister( PT7C4339_REG_A2_HOURS );
133 bool A2Is12H = hoursA2 & 0x40;
134 if( A2Is12H )
135 {
136 hoursA2 &= 0xBF;
137 if( !writeRegister( PT7C4339_REG_A2_HOURS, hoursA2 ) ) return 0;
138 }
139
140 if( getRtcStopFlag() ) return 2;
141 else return 1;
142}

183{
184 uint8_t registerData;
185
186 _i2cWire->beginTransmission( _i2cAddress );
187 _i2cWire->write( REG );
188 _i2cWire->endTransmission();
189
190 _i2cWire->requestFrom( _i2cAddress, static_cast<uint8_t>(1) );
191 while( _i2cWire->available() )
192 {
193 registerData = _i2cWire->read();
194 }
195

333PT7C4339_daysOfWeek PT7C4339::calculateWeekDay( uint16_t year, uint8_t month, uint8_t day )
334{
335 uint8_t yearCode;
336 uint8_t monthCode;
337 uint8_t centuryCode;
338 bool leapYearMod = false;
339
340 uint8_t yearLastTwoDigits = year % 100;
341 yearCode = ( ( yearLastTwoDigits % 100 ) + ( ( yearLastTwoDigits % 100 ) / 4 ) ) % 7;
342
343 bool isLeapYear = ( year % 4 == 0 && ( year % 100 != 0 || year % 400 == 0 ) );
344
345 switch ( month )
346 {
347 case 1:
348 monthCode = 0;
349 if( isLeapYear ) leapYearMod = true;
350 break;
351 case 2:
352 monthCode = 3;
353 if( isLeapYear ) leapYearMod = true;
354 break;
355 case 3:
356 monthCode = 3;
357 break;
358 case 4:
359 monthCode = 6;
360 break;
361 case 5:
362 monthCode = 1;
363 break;
364 case 6:
365 monthCode = 4;
366 break;
367 case 7:
368 monthCode = 6;
369 break;
370 case 8:
371 monthCode = 2;
372 break;
373 case 9:
374 monthCode = 5;
375 break;
376 case 10:
377 monthCode = 0;
378 break;
379 case 11:
380 monthCode = 3;
381 break;
382 case 12:
383 monthCode = 5;
384 break;
385 }
386
387 bool is2000;
388
389 if( year < 2000 ) is2000 = false;
390 else is2000 = true;
391
392 if( is2000 ) centuryCode = 6;
393 else centuryCode = 0;
394
395 uint8_t offsetWeekday = ( yearCode + monthCode + centuryCode + day - leapYearMod ) % 7;
396 if (offsetWeekday == 0) offsetWeekday = 7;
397
398 PT7C4339_daysOfWeek calculatedWeekDay = static_cast<PT7C4339_daysOfWeek>( offsetWeekday );
399
400 return calculatedWeekDay;
401}

676bool PT7C4339::setDay( uint8_t day )
677{
678 bool setSuccess;
679
680 uint8_t monthLength;
681
682 PT7C4339_Date oldDate = getDate();
683
684 bool isLeapYear = ( oldDate.year % 4 == 0 && ( oldDate.year % 100 != 0 || oldDate.year % 400 == 0 ) );
685
686 switch( oldDate.month )
687 {
688 case 1:
689 monthLength = 31;
690 break;
691 case 2:
692 if( !isLeapYear ) monthLength = 28;
693 else monthLength = 29;
694 break;
695 case 3: monthLength = 31; break;
696 case 4: monthLength = 30; break;
697 case 5: monthLength = 31; break;
698 case 6: monthLength = 30; break;
699 case 7: monthLength = 31; break;
700 case 8: monthLength = 31; break;
701 case 9: monthLength = 30; break;
702 case 10: monthLength = 31; break;
703 case 11: monthLength = 30; break;
704 case 12: monthLength = 31; break;
705 }
706
707 if( day <= monthLength && day > 0 )
708 {
709 day = decToBcd( day );
710 uint8_t oldDay = decToBcd( oldDate.day );
711
712 if( writeRegister( PT7C4339_REG_DATES, day ) && setCorrectWeekDay() ) setSuccess = true;
713 else
714 {
715 setSuccess = false;
716 writeRegister( PT7C4339_REG_DATES, oldDay );
717 setCorrectWeekDay();
718 }
719 }
720 else setSuccess = false;
721
722 return setSuccess;

776bool PT7C4339::setYear( uint16_t year )
777{
778 bool setSuccess;
779
780 uint8_t oldYear = getYear();
781
782 uint8_t oldMonth = getMonth();
783 uint8_t newMonth = decToBcd( oldMonth );
784 newMonth &= 0x7F;
785 oldMonth = newMonth;
786
787 if( year > 1899 && year < 2100 )
788 {
789 if( year > 1999 )
790 {
791 newMonth |= 0x80;
792 year -= 2000;
793 }
794 else year -= 1900;
795
796 if( oldYear > 1999 )
797 {
798 oldMonth |= 0x80;
799 oldYear -= 2000;
800 }
801 else oldYear -= 1900;
802
803 year = decToBcd( year );
804 oldYear = decToBcd( oldYear );
805
806 if( writeRegister( PT7C4339_REG_YEARS, year ) )
807 {
808 if( writeRegister( PT7C4339_REG_MONTHS, newMonth ) && setCorrectWeekDay() ) setSuccess = true;
809 else
810 {
811 setSuccess = false;
812 writeRegister( PT7C4339_REG_MONTHS, oldMonth );
813 writeRegister( PT7C4339_REG_YEARS, oldYear );
814 setCorrectWeekDay();
815 }
816 }
817 else
818 {
819 setSuccess = false;
820 writeRegister( PT7C4339_REG_YEARS, oldYear );
821 setCorrectWeekDay();
822 }
823 }
824 else setSuccess = false;
825
826 return setSuccess;
